359 template<typename DataType>
360 std::string * RFIm::getTimeDomainSigmaCutOpenCL_FrequencyTime_ReplaceWithMean(const RFImConfig & config, const std::string & dataTypeName, const AstroData::Observation & observation, const float sigmaCut, const unsigned int padding)
361 {
362  std::string *code = new std::string();
363  // Kernel template
364  *code = "__kernel void timeDomainSigmaCut(__global " + dataTypeName + " * const restrict time_series) {\n"
365  + config.getIntType() + " threshold = 0;\n"
366  "float delta = 0.0f;\n"
367  "float mean = 0.0f;\n"
368  "float sigma_cut = 0.0f;\n"
369  + dataTypeName + " sample_value;\n"
370  "__local float reductionCOU[" + std::to_string(config.getNrThreadsD0()) + "];\n"
371  "__local float reductionMEA[" + std::to_string(config.getNrThreadsD0()) + "];\n"
372  "__local float reductionVAR[" + std::to_string(config.getNrThreadsD0()) + "];\n"
373  "<%LOCAL_VARIABLES%>"
374  "\n"
375  "// Compute mean and standard deviation\n"
376  "for ( " + config.getIntType() + " sample_id = get_local_id(0) + " + std::to_string(config.getNrThreadsD0() * config.getNrItemsD0()) + "; sample_id < " + std::to_string(observation.getNrSamplesPerDispersedBatch(config.getSubbandDedispersion())) + "; sample_id += " + std::to_string(config.getNrThreadsD0() * config.getNrItemsD0()) + " ) "
377  "{\n"
378  "<%LOCAL_COMPUTE%>"
379  "}\n"
380  "// Local reduction (if necessary)\n"
381  "<%LOCAL_REDUCE%>"
382  "reductionCOU[get_local_id(0)] = counter_0;\n"
383  "reductionMEA[get_local_id(0)] = mean_0;\n"
384  "reductionVAR[get_local_id(0)] = variance_0;\n"
385  "barrier(CLK_LOCAL_MEM_FENCE);\n"
386  "// Global reduction\n"
387  "threshold = " + std::to_string(config.getNrThreadsD0() / 2) + ";\n"
388  "for (" + config.getIntType() + " sample_id = get_local_id(0); threshold > 0; threshold /= 2 ) {\n"
389  "if ( (sample_id < threshold)) {\n"
390  "delta = reductionMEA[sample_id + threshold] - mean_0;\n"
391  "counter_0 += reductionCOU[sample_id + threshold];\n"
392  "mean_0 = ((reductionCOU[sample_id] * mean_0) + (reductionCOU[sample_id + threshold] * reductionMEA[sample_id + threshold])) / counter_0;\n"
393  "variance_0 += reductionVAR[sample_id + threshold] + ((delta * delta) * ((reductionCOU[sample_id] * reductionCOU[sample_id + threshold]) / counter_0));\n"
394  "reductionCOU[sample_id] = counter_0;\n"
395  "reductionMEA[sample_id] = mean_0;\n"
396  "reductionVAR[sample_id] = variance_0;\n"
397  "}\n"
398  "barrier(CLK_LOCAL_MEM_FENCE);\n"
399  "}\n"
400  "mean = reductionMEA[0];\n"
401  "sigma_cut = (" + std::to_string(sigmaCut) + " * native_sqrt(reductionVAR[0] * " + std::to_string(1.0f/(observation.getNrSamplesPerDispersedBatch(config.getSubbandDedispersion()) - 1)) + "f));\n"
402  "// Replace samples over the sigma cut with mean\n"
403  "for (" + config.getIntType() + " sample_id = get_local_id(0); sample_id < " + std::to_string(observation.getNrSamplesPerDispersedBatch(config.getSubbandDedispersion())) + "; sample_id += " + std::to_string(config.getNrThreadsD0() * config.getNrItemsD0()) + " ) "
404  "{\n"
405  "<%REPLACE%>"
406  "}\n"
407  "}\n";
408  // Declaration of per thread variables
409  std::string localVariablesTemplate = "float counter_<%ITEM_NUMBER%> = 1.0f;\n"
410  "float variance_<%ITEM_NUMBER%> = 0.0f;\n"
411  "float mean_<%ITEM_NUMBER%> = time_series[(get_global_id(2) * " + std::to_string(observation.getNrChannels() * observation.getNrSamplesPerDispersedBatch(config.getSubbandDedispersion(), padding / sizeof(DataType))) + ") + (get_global_id(1) * " + std::to_string(observation.getNrSamplesPerDispersedBatch(config.getSubbandDedispersion(), padding / sizeof(DataType))) + ") + get_local_id(0) + <%ITEM_OFFSET%>];\n";
412  // Local compute
413  // Version without boundary checks
414  std::string localComputeNoCheckTemplate = "sample_value = time_series[(get_global_id(2) * " + std::to_string(observation.getNrChannels() * observation.getNrSamplesPerDispersedBatch(config.getSubbandDedispersion(), padding / sizeof(DataType))) + ") + (get_global_id(1) * " + std::to_string(observation.getNrSamplesPerDispersedBatch(config.getSubbandDedispersion(), padding / sizeof(DataType))) + ") + get_local_id(0) + <%ITEM_OFFSET%>];\n"
415  "counter_<%ITEM_NUMBER%> += 1.0f;\n"
416  "delta = sample_value - mean_<%ITEM_NUMBER%>;\n"
417  "mean_<%ITEM_NUMBER%> += delta / counter_<%ITEM_NUMBER%>;\n"
418  "variance_<%ITEM_NUMBER%> += delta * (sample_value - mean_<%ITEM_NUMBER%>);\n";
419  // Version with boundary checks
420  std::string localComputeCheckTemplate = "if ( sample_id + <%ITEM_OFFSET%> < " + std::to_string(observation.getNrSamplesPerDispersedBatch(config.getSubbandDedispersion())) + " ) {\n"
421  "sample_value = time_series[(get_global_id(2) * " + std::to_string(observation.getNrChannels() * observation.getNrSamplesPerDispersedBatch(config.getSubbandDedispersion(), padding / sizeof(DataType))) + ") + (get_global_id(1) * " + std::to_string(observation.getNrSamplesPerDispersedBatch(config.getSubbandDedispersion(), padding / sizeof(DataType))) + ") + get_local_id(0) + <%ITEM_OFFSET%>];\n"
422  "counter_<%ITEM_NUMBER%> += 1.0f;\n"
423  "delta = sample_value - mean_<%ITEM_NUMBER%>;\n"
424  "mean_<%ITEM_NUMBER%> += delta / counter_<%ITEM_NUMBER%>;\n"
425  "variance_<%ITEM_NUMBER%> += delta * (sample_value - mean_<%ITEM_NUMBER%>);\n"
426  "}\n";
427  // In-thread reduction
428  std::string localReduceTemplate = "delta = mean_<%ITEM_NUMBER%> - mean_0;\n"
429  "counter_0 += counter_<%ITEM_NUMBER%>;\n"
430  "mean_0 = (((counter_0 - counter_<%ITEM_NUMBER%>) * mean_0) + (counter_<%ITEM_NUMBER%> * mean_<%ITEM_NUMBER%>)) / counter_0;\n"
431  "variance_0 += variance_<%ITEM_NUMBER%> + ((delta * delta) * (((counter_0 - counter_<%ITEM_NUMBER%>) * counter_<%ITEM_NUMBER%>) / counter_0));\n";
432  std::string replaceConditionTemplate = "if ( time_series[(get_global_id(2) * " + std::to_string(observation.getNrChannels() * observation.getNrSamplesPerDispersedBatch(config.getSubbandDedispersion(), padding / sizeof(DataType))) + ") + (get_global_id(1) * " + std::to_string(observation.getNrSamplesPerDispersedBatch(config.getSubbandDedispersion(), padding / sizeof(DataType))) + ") + get_local_id(0) + <%ITEM_OFFSET%>] > (mean + sigma_cut) ) {\n"
433  "time_series[(get_global_id(2) * " + std::to_string(observation.getNrChannels() * observation.getNrSamplesPerDispersedBatch(config.getSubbandDedispersion(), padding / sizeof(DataType))) + ") + (get_global_id(1) * " + std::to_string(observation.getNrSamplesPerDispersedBatch(config.getSubbandDedispersion(), padding / sizeof(DataType))) + ") + get_local_id(0) + <%ITEM_OFFSET%>] = mean;"
434  "}\n";
435  std::string replaceTemplate = "sample_value = time_series[(get_global_id(2) * " + std::to_string(observation.getNrChannels() * observation.getNrSamplesPerDispersedBatch(config.getSubbandDedispersion(), padding / sizeof(DataType))) + ") + (get_global_id(1) * " + std::to_string(observation.getNrSamplesPerDispersedBatch(config.getSubbandDedispersion(), padding / sizeof(DataType))) + ") + get_local_id(0) + <%ITEM_OFFSET%>];\n"
436  "time_series[(get_global_id(2) * " + std::to_string(observation.getNrChannels() * observation.getNrSamplesPerDispersedBatch(config.getSubbandDedispersion(), padding / sizeof(DataType))) + ") + (get_global_id(1) * " + std::to_string(observation.getNrSamplesPerDispersedBatch(config.getSubbandDedispersion(), padding / sizeof(DataType))) + ") + get_local_id(0) + <%ITEM_OFFSET%>] = (sample_value * (convert_" + dataTypeName + "(sample_value < (mean + sigma_cut)))) + (mean * (convert_" + dataTypeName + "(sample_value > mean + sigma_cut)));\n";
437  // End of kernel template
438  // Code generation
439  std::string localVariables;
440  std::string localCompute;
441  std::string localReduce;
442  std::string replace;
443  for ( unsigned int item = 0; item < config.getNrItemsD0(); item++ )
444  {
445  std::string * temp;
446  std::string itemString = std::to_string(item);
447  std::string itemOffsetString = std::to_string(item * config.getNrThreadsD0());
448  temp = isa::utils::replace(&localVariablesTemplate, "<%ITEM_NUMBER%>", itemString);
449  if ( item == 0 )
450  {
451  temp = isa::utils::replace(temp, " + <%ITEM_OFFSET%>", std::string(), true);
452  }
453  else
454  {
455  temp = isa::utils::replace(temp, "<%ITEM_OFFSET%>", itemOffsetString, true);
456  }
457  localVariables.append(*temp);
458  delete temp;
459  if ( (observation.getNrSamplesPerDispersedBatch(config.getSubbandDedispersion()) % (config.getNrThreadsD0() * config.getNrItemsD0())) == 0 )
460  {
461  temp = isa::utils::replace(&localComputeNoCheckTemplate, "<%ITEM_NUMBER%>", itemString);
462  }
463  else
464  {
465  temp = isa::utils::replace(&localComputeCheckTemplate, "<%ITEM_NUMBER%>", itemString);
466  }
467  if ( item == 0 )
468  {
469  temp = isa::utils::replace(temp, " + <%ITEM_OFFSET%>", std::string(), true);
470  }
471  else
472  {
473  temp = isa::utils::replace(temp, "<%ITEM_OFFSET%>", itemOffsetString, true);
474  }
475  localCompute.append(*temp);
476  delete temp;
477  if ( item > 0 )
478  {
479  temp = isa::utils::replace(&localReduceTemplate, "<%ITEM_NUMBER%>", itemString);
480  localReduce.append(*temp);
481  delete temp;
482  }
483  if ( config.getConditionalReplacement() )
484  {
485  temp = isa::utils::replace(&replaceConditionTemplate, "<%ITEM_NUMBER%>", itemString);
486  }
487  else
488  {
489  temp = isa::utils::replace(&replaceTemplate, "<%ITEM_NUMBER%>", itemString);
490  }
491  if ( item == 0 )
492  {
493  temp = isa::utils::replace(temp, " + <%ITEM_OFFSET%>", std::string(), true);
494  }
495  else
496  {
497  temp = isa::utils::replace(temp, "<%ITEM_OFFSET%>", itemOffsetString, true);
498  }
499  replace.append(*temp);
500  }
501  code = isa::utils::replace(code, "<%LOCAL_VARIABLES%>", localVariables, true);
502  code = isa::utils::replace(code, "<%LOCAL_COMPUTE%>", localCompute, true);
503  code = isa::utils::replace(code, "<%LOCAL_REDUCE%>", localReduce, true);
504  code = isa::utils::replace(code, "<%REPLACE%>", replace, true);
505  return code;
506 }

757 std::uint64_t RFIm::frequencyDomainSigmaCut(const bool subbandDedispersion, const DataOrdering & ordering, const ReplacementStrategy & replacement, const AstroData::Observation & observation, std::vector<DataType> & time_series, const unsigned int nrBins, const float sigmaCut, const unsigned int padding)
758 {
759  std::uint64_t replacedSamples = 0;
760  unsigned int nrChannelsPerBin = static_cast<unsigned int>(std::ceil(observation.getNrChannels() / nrBins));
761  for ( unsigned int beam = 0; beam < observation.getNrBeams(); beam++ )
762  {
763  if ( ordering == FrequencyTime )
764  {
765  for ( unsigned int sample_id = 0; sample_id < observation.getNrSamplesPerDispersedBatch(subbandDedispersion); sample_id++ )
766  {
767  isa::utils::Statistics<DataType> statistics_corrected;
768  isa::utils::Statistics<DataType> * local_statistics = new isa::utils::Statistics<DataType> [nrBins];
769  for ( unsigned int bin = 0; bin < nrBins; bin++ )
770  {
771  // Compute the statistics for a bin
772  for ( unsigned int channel = bin * nrChannelsPerBin; channel < (bin + 1) * nrChannelsPerBin; channel++ )

774  local_statistics[bin].addElement(time_series.at((beam * observation.getNrChannels() * observation.getNrSamplesPerDispersedBatch(subbandDedispersion, padding / sizeof(DataType))) + (channel * observation.getNrSamplesPerDispersedBatch(subbandDedispersion, padding / sizeof(DataType))) + sample_id));
775  }

777  for ( unsigned int channel = bin * nrChannelsPerBin; channel < (bin + 1) * nrChannelsPerBin; channel++ )
778  {

780  }

786  {

794  }
795  }
796  }
797  }
798  delete [] local_statistics;
799  }
800  }
801  }
802  return replacedSamples;
803 }
